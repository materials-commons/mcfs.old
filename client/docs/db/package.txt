    ProjectEvents *model.Query

    // ProjectFiles is the query model for project files
    ProjectFiles *model.Query
)


FUNCTIONS

func Exists() bool
    Exists returns true if the database exists. This is determined by
    looking for a database file entry.

func Open() error
    Open sets up the models and the database. If the database doesn't exist
    then Open will create the database and it's schema.

func Path() string
    Path returns the full path to the database file.

func Use(db *sqlx.DB)
    Use sets the database connection for all the models.
